/// Maximum connection slots per master.
pub const MAX_CONNECT_SLOTS: u8 = 16;

/// A named connection slot for a specific master.
///
/// Each slot has its own bunker URI secret. The slot -- not the ephemeral
/// client pubkey -- is the stable identity. When a client reconnects with
/// a new ephemeral key, the slot's pubkey is swapped and permissions carry over.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ConnectSlot {
    /// Slot index (0-15).
    pub slot_index: u8,
    /// Human-readable label (e.g. "nostrudel desktop").
    pub label: String,
    /// Hex-encoded 32-byte secret (64 chars). Generated by hardware RNG.
    pub secret: String,
    /// Hex-encoded current client pubkey, set on first connect.
    #[serde(default)]
    pub current_pubkey: Option<String>,
    /// Which NIP-46 methods are auto-approved for this slot.
    #[serde(default)]
    pub allowed_methods: Vec<String>,
    /// Which event kinds are auto-approved for sign_event.
    #[serde(default)]
    pub allowed_kinds: Vec<u64>,
    /// Whether to auto-approve matching requests (true) or just OLED-notify (false).
    #[serde(default)]
    pub auto_approve: bool,
    /// True once the user has physically approved a sign_event via button press.
    #[serde(default)]
    pub signing_approved: bool,
}
```

/// Find a slot by its current client pubkey.
pub fn find_slot_by_pubkey<'a>(slots: &'a [ConnectSlot], pubkey: &str) -> Option<&'a ConnectSlot> {
    slots.iter().find(|s| s.current_pubkey.as_deref() == Some(pubkey))
}

/// Find a mutable slot by its current client pubkey.
pub fn find_slot_by_pubkey_mut<'a>(slots: &'a mut [ConnectSlot], pubkey: &str) -> Option<&'a mut ConnectSlot> {
    slots.iter_mut().find(|s| s.current_pubkey.as_deref() == Some(pubkey))
}

/// Find a slot by its secret using constant-time comparison.
/// Always iterates all slots to prevent timing-based slot identification.
pub fn find_slot_by_secret<'a>(slots: &'a [ConnectSlot], secret: &str) -> Option<&'a ConnectSlot> {
    let mut matched: Option<usize> = None;
    for (i, slot) in slots.iter().enumerate() {
        if constant_time_eq_str(&slot.secret, secret) {
            matched = Some(i);
        }
    }
    matched.map(|i| &slots[i])
}

/// Find a mutable slot by its secret using constant-time comparison.
pub fn find_slot_by_secret_mut<'a>(slots: &'a mut [ConnectSlot], secret: &str) -> Option<&'a mut ConnectSlot> {
    let mut matched: Option<usize> = None;
    for (i, slot) in slots.iter().enumerate() {
        if constant_time_eq_str(&slot.secret, secret) {
            matched = Some(i);
        }
    }
    matched.map(move |i| &mut slots[i])
}

/// Find the next available slot index (0-15). Returns None if all 16 are occupied.
pub fn next_slot_index(slots: &[ConnectSlot]) -> Option<u8> {
    for i in 0..MAX_CONNECT_SLOTS {
        if !slots.iter().any(|s| s.slot_index == i) {
            return Some(i);
        }
    }
    None
}

/// Create a redacted copy of a slot (secret replaced with empty string).
pub fn redact_slot(slot: &ConnectSlot) -> ConnectSlot {
    ConnectSlot {
        secret: String::new(),
        ..slot.clone()
    }
}

/// Constant-time string equality (prevents timing attacks on secret comparison).
fn constant_time_eq_str(a: &str, b: &str) -> bool {
    let a = a.as_bytes();
    let b = b.as_bytes();
    if a.len() != b.len() {
        return false;
    }
    let mut diff = 0u8;
    for (x, y) in a.iter().zip(b.iter()) {
        diff |= x ^ y;
    }
    diff == 0
}
```

/// Determine the approval tier for a request.
pub fn check(
    &self,
    master_slot: u8,
    client_pubkey: &str,
    method: &Nip46Method,
    event_kind: Option<u64>,
) -> ApprovalTier {
    let slot = self.master_slots.iter()
        .find(|ms| ms.master_slot == master_slot)
        .and_then(|ms| heartwood_common::policy::find_slot_by_pubkey(&ms.slots, client_pubkey));

    let slot = match slot {
        Some(s) => s,
        None => return ApprovalTier::ButtonRequired,
    };

    // Unknown method -- always require button.
    let method_str = method.as_str();
    if !slot.allowed_methods.iter().any(|m| m == method_str) {
        return ApprovalTier::ButtonRequired;
    }

    // sign_event with kind restrictions.
    if matches!(method, Nip46Method::SignEvent) {
        if !slot.allowed_kinds.is_empty() {
            if let Some(kind) = event_kind {
                if !slot.allowed_kinds.contains(&kind) {
                    return ApprovalTier::ButtonRequired;
                }
            } else {
                // Can't determine kind -- require button.
                return ApprovalTier::ButtonRequired;
            }
        }
    }

    if slot.auto_approve {
        ApprovalTier::AutoApprove
    } else {
        ApprovalTier::OledNotify
    }
}

/// Update a slot's label, methods, kinds, or auto_approve.
pub fn update_slot(
    &mut self,
    master_slot: u8,
    slot_index: u8,
    label: Option<String>,
    allowed_methods: Option<Vec<String>>,
    allowed_kinds: Option<Vec<u64>>,
    auto_approve: Option<bool>,
) -> bool {

    if let Some(slot) = slots.iter_mut().find(|s| s.slot_index == slot_index) {

    }

/// Revoke (delete) a slot by index.
pub fn revoke_slot(&mut self, master_slot: u8, slot_index: u8) -> bool {

    let removed = slots.len() < before;
    if removed {

        }

        false
    }
}

/// Persist all slots for a master to NVS if changed since last write.
pub fn persist_slots(&mut self, nvs: &mut EspNvs<NvsDefault>, master_slot: u8) {
    if !self.slots_dirty {
        return;
    }
    let ms = self.master_slots.iter().find(|ms| ms.master_slot == master_slot);
    let key = format!("connslots_{master_slot}");
    match ms {
        Some(ms) => {
            match serde_json::to_string(&ms.slots) {
                Ok(json) => {
                    if let Err(e) = nvs.set_blob(&key, json.as_bytes()) {
                        log::error!("Failed to persist slots for master {master_slot}: {e:?}");
                    }
                }
                Err(e) => log::error!("Failed to serialise slots: {e}"),
            }
        }
        None => { let _ = nvs.remove(&key); }
    }
    self.slots_dirty = false;
}

/// Load slots from NVS for all masters. Migrates old policy format if found.
pub fn load_from_nvs(nvs: &EspNvs<NvsDefault>, master_count: u8) -> Self {
    let mut engine = Self::new();
    for slot in 0..master_count {
        // Try new format first.
        let new_key = format!("connslots_{slot}");
        let mut buf = [0u8; 8192];
        if let Ok(Some(data)) = nvs.get_blob(&new_key, &mut buf) {
            if let Ok(slots) = serde_json::from_slice::<Vec<ConnectSlot>>(data) {
                let count = slots.len();
                engine.master_slots.push(MasterSlots { master_slot: slot, slots });
                log::info!("Loaded {count} connection slots for master {slot}");
                continue;
            }
        }

        // Migration: check for old policy format.
        let old_policy_key = format!("policy_{slot}");
        let old_conn_key = format!("master_{slot}_conn");
        let mut old_secret = [0u8; 32];
        let has_old_secret = matches!(
            nvs.get_blob(&old_conn_key, &mut old_secret),
            Ok(Some(b)) if b.len() == 32
        );

        if has_old_secret {
            let secret_hex = heartwood_common::hex::hex_encode(&old_secret);
            let migrated = ConnectSlot {
                slot_index: 0,
                label: "default".to_string(),
                secret: secret_hex,
                current_pubkey: None,
                allowed_methods: heartwood_common::policy::CONNECT_SAFE_METHODS
                    .iter().map(|s| s.to_string()).collect(),
                allowed_kinds: vec![],
                auto_approve: true,
                signing_approved: false,
            };
            engine.master_slots.push(MasterSlots {
                master_slot: slot,
                slots: vec![migrated],
            });
            engine.slots_dirty = true;
            log::info!("Migrated old connect secret for master {slot} to default connection slot");

            // Clean up old keys.
            let _ = nvs.remove(&old_policy_key);
            // Note: don't remove old_conn_key yet -- masters.rs still reads it.
            // It will be cleaned up in Task 5 when LoadedMaster drops connect_secret.
        }
    }
    engine
}

    if !policy_engine.bridge_authenticated {
        protocol::write_frame(&mut usb, FRAME_TYPE_NACK, &[]);
    } else if frame.payload.is_empty() {
        log::warn!("CONNSLOT_CREATE missing master_slot");
        protocol::write_frame(&mut usb, FRAME_TYPE_NACK, &[]);
    } else {
        let ms = frame.payload[0];
        let label = if frame.payload.len() > 1 {
            String::from_utf8_lossy(&frame.payload[1..]).to_string()

        unsafe {
            esp_idf_svc::sys::esp_fill_random(
